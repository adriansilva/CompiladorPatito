PROGRAMA p;
VAR INT: arr[10];

FUNCION INT binarySearch(INT arr[10], INT x, INT low, INT high){
    VAR INT: mid = (low+high+1)/2;

    SI (low == high-1) ENTONCES{
      SI (low == -1) ENTONCES{
          REGRESA low;
      }
      SI (arr[low] == x) ENTONCES{
        REGRESA low;
      }
      SINO{
        REGRESA -1;
      }
    }

    SI(arr[mid]>x)ENTONCES{
      high = mid;
    }
    SINO{
      low = mid;
    }
    REGRESA binarySearch(arr,x,low,high);
}
